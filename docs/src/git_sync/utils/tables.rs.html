19use console::{Term, measure_text_width};
20use std::fmt;
21use tabled::{
22    builder::Builder,
23    settings::{Alignment, Panel, style::Style},
24};
25
26/// A struct wrapping around tabled's tables so that we can build tables more easily
27pub struct Table<T, B, L, R, H, V, const HSIZE: usize, const VSIZE: usize> {
28    header: Vec<String>,
29    rows: Vec<Vec<String>>,
30    title: Option<String>,
31    style: Style<T, B, L, R, H, V, HSIZE, VSIZE>,
32    alignment: Option<Alignment>,
33    centre: bool,
34    inner_table_: tabled::Table,
35}
36
37impl<
38    T: Clone,
39    B: Clone,
40    L: Clone,
41    R: Clone,
42    H: Clone,
43    V: Clone,
44    const HSIZE: usize,
45    const VSIZE: usize,
46> Table<T, B, L, R, H, V, HSIZE, VSIZE>
47{
48    pub fn builder(style: Style<T, B, L, R, H, V, HSIZE, VSIZE>) -> Self {
49        Table {
50            header: Vec::new(),
51            rows: Vec::new(),
52            title: None,
53            style,
54            alignment: None,
55            centre: false,
56            inner_table_: tabled::Table::default(),
57        }
58    }
59    /// Add a header to the table
60    pub fn header<I, S>(mut self, header: I) -> Self
61    where
62        I: IntoIterator<Item = S>,
63        S: Into<String>,
64    {
65        self.header = header.into_iter().map(Into::into).collect();
66        self
67    }
68
69    /// Set the rows for the table
70    pub fn rows<I, S>(mut self, rows: I) -> Self
71    where
72        I: IntoIterator,
73        I::Item: IntoIterator<Item = S>,
74        S: Into<String>,
75    {
76        self.rows = rows
77            .into_iter()
78            .map(|row| row.into_iter().map(Into::into).collect())
79            .collect();
80
81        self
82    }
83    /// Add a title to your table
84    pub fn title<S: Into<String>>(mut self, title: S) -> Self {
85        self.title = Some(title.into());
86        self
87    }
88    /// Set the alignment of the row/column
89    pub fn align(mut self, alignment: Alignment) -> Self {
90        self.alignment = Some(alignment);
91        self
92    }
93    /// Define if the table should be centred in the terminal
94    pub fn centre(mut self, centre: bool) -> Self {
95        self.centre = centre;
96        self
97    }
98    /// Build the table
99    pub fn build(mut self) -> Self {
100        // Use tabled's builder pattern
101        let mut builder = Builder::default();
102        if !self.header.is_empty() {
103            builder.push_record(&self.header);
104        }
105        for row in &self.rows {
106            builder.push_record(row);
107        }
108        let mut table = builder.build();
109        table.with(self.style.clone());
110
111        if let Some(ref title) = self.title {
112            table.with(Panel::horizontal(0, title));
113        }
114
115        if let Some(alignment) = self.alignment {
116            table.with(alignment);
117        }
118        self.inner_table_ = table;
119
120        self
121    }
122}
123
124impl<
125    T: fmt::Debug,
126    B: fmt::Debug,
127    L: fmt::Debug,
128    R: fmt::Debug,
129    H: fmt::Debug,
130    V: fmt::Debug,
131    const HSIZE: usize,
132    const VSIZE: usize,
133> fmt::Display for Table<T, B, L, R, H, V, HSIZE, VSIZE>
134{
135    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
136        let term = Term::stdout();
137
138        let table = if self.rows.is_empty() {
139            String::new()
140        } else if self.centre && term.is_term() {
141            let term_width = term.size().1 as usize;
142            self.inner_table_
143                .to_string()
144                .lines()
145                .map(|line| {
146                    let line_width = measure_text_width(line);
147                    let padding = if term_width > line_width {
148                        (term_width - line_width) / 2
149                    } else {
150                        0
151                    };
152                    let padding_amount = " ".repeat(padding);
153                    format!("{padding_amount}{line}{padding_amount}")
154                })
155                .collect::<Vec<String>>()
156                .join("\n")
157        } else if term.is_term() {
158            self.inner_table_.to_string()
159        } else {
160            let mut output = String::new();
161            for row in &self.rows {
162                let line = row.join(",");
163                output.push_str(&line);
164                output.push('\n');
165            }
166            output
167        };
168
169        write!(f, "{table}")
170    }
171}
