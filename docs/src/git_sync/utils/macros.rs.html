20/// Handles the result of an API call using octocrab and maps them
21/// to a `GitError`. This helps reduce a lot of unnecessary boilerplate.
22/// This is used for api calls that don't already have a high level
23/// function provided by Octocrab.
24///
25/// # Parameters
26/// * `$response` -- The result to match.
27/// * `$context` -- A string describing the context of what you're trying to handle, for logging.
28/// * `$ok` -- The closure to run if the result is Ok.
29///
30/// # Returns
31/// Returns the result of `$ok` if successful, or a `GitError` if the API call fails.
32/// You can use '?' to propagate the error to your function.
33///
34/// ```rust
35/// async fn sync_fork(owner: &str, repo: &str, parent_owner: &str, parent_repo: &str, main_branch: &str) -> Result<(), GitError> {
36///     let octocrab = octocrab::instance();
37///     let body = serde_json::json!({"branch": main_branch.to_string()});
38///     let response: Result<serde_json::Value, octocrab::Error> = octocrab
39///         .post(format!("/repos/{owner}/{repo}/merge-upstream"), Some(&body))
40///         .await;
41///
42///     handle_api_response!(
43///         response,
44///         format!("Unable to sync {owner}/{repo} with {parent_owner}/{parent_repo}"),
45///         |_| {
46///             println!("Successfully synced {owner}/{repo} with {parent_owner}/{parent_repo}");
47///             Ok::<(), GitError>(())
48///         })?;
49///        Ok(())
50///    }
51/// ```
52#[macro_export]
53macro_rules! handle_api_response {
54    (
55        $response:expr,
56        $context:expr,
57        $ok:expr,
58    ) => {
59        match $response {
60            Ok(body) => $ok(body),
61            Err(err) => {
62                let (status, message) = $crate::utils::repo::get_http_status(&err);
63                match (status, message) {
64                    (Some(code), Some(message)) => {
65                        eprintln!("{}: HTTP {:?} - {:?}", $context, code, message)
66                    }
67                    (Some(code), None) => eprintln!("{}: HTTP {:?}", $context, code),
68                    _ => eprintln!("{}: {:?}", $context, err),
69                }
70                return Err(GitError::GithubApiError(err));
71            }
72        }
73    };
74}

76/// Handles reducing some of the boiler plate for defining these unordered futures.
77///
78/// # Parameters
79/// `$iter` is some iterable collection
80/// `|$($arg:ident),*| $future:expr`: Closure-like syntax where you specify the variable(s)
81/// `($($pat:pat),*) $body:block`: Pattern(s) to destructure the resulting tuple from the completed future
82/// and a block of code to process each result.
83///
84/// # Example
85/// ```rust
86/// handle_futures_unordered!(
87///     vec.iter().map(|x| (x.clone(),)), |x| async move {
88///         (x, async_function(x).await)
89///     },
90///     (x, result) {
91///         match result {
92///             Ok(val) => println!("Success: {x}: {val}"),
93///             Err(e) => eprintln!("Error: {x}: {e}"),
94///         }
95///     }
96/// );
97/// pub async fn sync_all_forks(&self, repos: Vec<String>) -> Result<(), GitError> {
98///     handle_futures_unordered!(
99///         repos.iter().map(|repo| (repo.clone(),)),
100///         |repo| self.sync_fork(&repo),
101///         (repo, result) {
102///             match result {
103///                 Ok(_) => println!("Successfully synced fork: {}", repo),
104///                 Err(e) => eprintln!("Error syncing fork {}: {}", repo, e),
105///             }
106///         }
107///     );
108/// }
109///
110/// ```
111/// # Notes
112/// - The macro requires the `futures` crate.
113/// - The async block must return a tuple matching the destructuring pattern used in the `while let Some(...)`.
114/// - All generated futures must have the same return type.
115///
116/// # See also
117/// - [`FuturesUnordered`](https://docs.rs/futures/latest/futures/stream/struct.FuturesUnordered.html)
118#[macro_export]
119macro_rules! handle_futures_unordered {
120    (
121        $iter:expr, |$($arg:ident),*| $future:expr, ($($pattern:pat),*) $body:expr
122    ) => {{
123        let mut futures = ::futures::stream::FuturesUnordered::new();
124        for ($($arg),*) in $iter {
125            futures.push(async move {$future.await});
126        }
127        while let Some(($($pattern),*)) = futures.next().await {
128            $body
129        }
130
131    }};
132}

134/// A macro to easily retry asynchronous operations with exponential backoff and optional error filtering.
135///
136/// # Overview
137/// The `async_retry!` macro wraps an asynchronous block and automatically retries it using an
138/// exponential backoff schedule. It allows you to specify the initial delay, maximum delay,
139/// number of retries, a custom error predicate, and the operation body.
140///
141/// Internally, it uses [`tokio_retry::RetryIf`] with [`tokio_retry::strategy::ExponentialBackoff`],
142/// applying jitter to the delays.
143///
144/// # Parameters
145/// - `ms = $ms`: The base delay in milliseconds for the exponential backoff.
146/// - `timeout = $timeout`: The maximum delay between retries, in milliseconds.
147/// - `retries = $retries`: The maximum number of retry attempts.
148/// - `error_predicate = $pred`: A function or closure that takes a reference to an error and returns
149///   `true` if the error is retryable, or `false` otherwise.
150/// - `body = $body`: The asynchronous block to execute and retry on failure.
151///
152/// # Example
153/// ```rust
154/// use tokio_retry::strategy::ExponentialBackoff;
155/// use tokio_retry::RetryIf;
156///
157/// async fn unreliable_network_call() -> Result<(), std::io::Error> {
158///     // ... implementation ...
159///     Ok(())
160/// }
161///
162/// let result: Result<(), std::io::Error> = async_retry!(
163///     ms = 100,
164///     timeout = 2000,
165///     retries = 5,
166///     error_predicate = |e: &std::io::Error| matches!(e.kind(), std::io::ErrorKind::TimedOut),
167///     body = {
168///         unreliable_network_call().await
169///     },
170/// );
171/// ```
172///
173/// # Notes
174/// - The macro requires the `tokio_retry` and `tokio` crates.
175/// - The `body` block must be `async`.
176/// - The retry strategy applies exponential backoff with jitter and is capped by the specified timeout.
177///
178/// # See Also
179/// - [`tokio_retry::RetryIf`](https://docs.rs/tokio-retry/latest/tokio_retry/struct.RetryIf.html)
180/// - [`tokio_retry::strategy::ExponentialBackoff`](https://docs.rs/tokio-retry/latest/tokio_retry/strategy/struct.ExponentialBackoff.html)
181#[macro_export]
182macro_rules! async_retry {
183    (
184        ms = $ms: expr,
185        timeout = $timeout: expr,
186        retries = $retries: expr,
187        error_predicate = $pred: expr,
188        body = $body: block,
189    ) => {{
190        let retry_strategy = tokio_retry::strategy::ExponentialBackoff::from_millis($ms)
191            .max_delay(tokio::time::Duration::from_millis($timeout))
192            .map(tokio_retry::strategy::jitter)
193            .take($retries);
194
195        tokio_retry::RetryIf::spawn(retry_strategy, || async { $body }, $pred).await
196    }};
