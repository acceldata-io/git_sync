20use crate::error::{GitError, is_retryable};
21use crate::github::client::GithubClient;
22use crate::utils::repo::get_repo_info_from_url;
23use crate::{async_retry, handle_api_response};
24use futures::{StreamExt, stream::FuturesUnordered};
25
26impl GithubClient {
27    /// Sync a single repository with its parent repository.
28    pub async fn sync_fork(&self, url: &str) -> Result<(), GitError> {
29        let info = get_repo_info_from_url(url)?;
30        let (owner, repo) = (info.owner, info.repo_name);
31        println!("Syncing {owner}/{repo} with its parent repository...");
32
33        let parent = self.get_parent_repo(url).await?;
34        let body = serde_json::json!({"branch": parent.main_branch});
35        let octocrab = self.octocrab.clone();
36        // Retry if a potentially recoverable error is detected
37        let response: Result<serde_json::Value, octocrab::Error> = async_retry!(
38            ms = 100,
39            timeout = 5000,
40            retries = 3,
41            error_predicate = |e: &octocrab::Error| is_retryable(e),
42            body = {
43                octocrab
44                    .post(format!("/repos/{owner}/{repo}/merge-upstream"), Some(&body))
45                    .await
46            },
47        );
48        handle_api_response!(
49            response,
50            format!("Unable to sync {owner}/{repo} with its parent repository"),
51            |_| {
52                println!("Successfully synced {owner}/{repo} with its parent repository");
53                Ok::<(), GitError>(())
54            },
55        )?;
56        Ok(())
57    }
58
59    /// Sync all configured repositories. Only repositories that have a parent repository
60    /// should be passed to this function
61    pub async fn sync_all_forks(&self, repositories: Vec<String>) -> Result<(), GitError> {
62        let mut futures = FuturesUnordered::new();
63        for repo in repositories {
64            let semaphore_permit = self.semaphore.clone().acquire_owned().await?;
65            futures.push(async move {
66                let result = self.sync_fork(&repo).await;
67                drop(semaphore_permit);
68                (repo, result)
69            });
70        }
71        let mut errors: Vec<(String, GitError)> = Vec::new();
72        while let Some((repo, result)) = futures.next().await {
73            match result {
74                Ok(()) => {
75                    println!("✅ Successfully synced {repo}");
76                }
77                Err(e) => {
78                    println!("❌ Failed to sync {repo}: {e}");
79                    errors.push((repo.to_string(), e));
80                }
81            }
82        }
83        if !errors.is_empty() {
84            return Err(GitError::MultipleErrors(errors));
85        }
86        Ok(())
87    }
88}
