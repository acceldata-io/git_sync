45    pub async fn backup_repo(&self, url: String, path: &Path) -> Result<(), GitError> {
46        let semaphore_lock = Arc::clone(&self.semaphore).acquire_owned().await?;
47        let canonical_path = if let Ok(p) = path.canonicalize() {
48            p
49        } else {
50            path.to_path_buf()
51        };
52
53        let ssh_url = http_to_ssh_repo(&url)?;
54
55        let Some(name) = ssh_url.split('/').next_back() else {
56            return Err(GitError::Other(format!(
57                "Failed to extract repository name from URL: {ssh_url}"
58            )));
59        };
60
61        let mut path = canonical_path.to_string_lossy().to_string();
62        path.push('/');
63        path.push_str(name);
64
65        if let Some(t) = self.output.get() {
66            if *t == OutputMode::Print && self.is_tty {
67                println!("Backing up repository {url} to path: {path}");
68            }
69        }
70        // If the mirrored repo already exists, update it rather than clone again
71        let result = if Path::new(&path).exists() {
72            tokio::task::spawn_blocking(move || {
73                let _lock = semaphore_lock;
74                // If the path exists, we assume it's a git repository and we fetch the latest changes
75                let output = Command::new("git")
76                    .stdout(Stdio::piped())
77                    .stderr(Stdio::piped())
78                    .args(["remote", "update"])
79                    .current_dir(&path)
80                    .output();
81                match output {
82                    // This branch happens only if the output of the command was successful
83                    Ok(s) if s.status.success() => Ok::<_, GitError>(format!(
84                        "Successfully backed up repository: {ssh_url} to path: {path}"
85                    )),
86                    Ok(s) => {
87                        let stderr = String::from_utf8_lossy(&s.stderr);
88                        eprintln!("Error while backing up.");
89                        eprintln!("STDERR:\n{stderr}");
90                        Err(GitError::Other(format!(
91                            "Failed to back up repository: {ssh_url}, git exited with status: {}",
92                            s.status
93                        )))
94                    }
95                    Err(e) => Err(GitError::Other(format!(
96                        "Failed to back up repository: {ssh_url}, error: {e}"
97                    ))),
98                }
99            })
100            .await?
101        // Mirror the repo
102        } else {
103            tokio::task::spawn_blocking(move || {
104                let _lock = semaphore_lock;
105                // Clone the reposittory as a mirror
106                let output = Command::new("git")
107                    .stdout(Stdio::piped())
108                    .stderr(Stdio::piped())
109                    .args(["clone", "--mirror", &ssh_url, &path])
110                    .output();
111                match output {
112                    // This branch happens only if the output of the command was successful
113                    Ok(s) if s.status.success() => Ok::<_, GitError>(format!(
114                        "Successfully backed up repository: {ssh_url} to path: {path}"
115                    )),
116                    Ok(s) => {
117                        if Path::new(&path).exists() {
118                            // If the clone failed, remove the directory to avoid leaving a broken repo
119                            std::fs::remove_dir_all(&path)?;
120                        }
121                        let stderr = String::from_utf8_lossy(&s.stderr);
122                        eprintln!("Error while backing up.");
123                        eprintln!("STDERR:\n{stderr}");
124                        Err(GitError::Other(format!(
125                            "Failed to back up repository: {ssh_url}, git exited with status: {}",
126                            s.status
127                        )))
128                    }
129                    Err(e) => Err(GitError::Other(format!(
130                        "Failed to back up repository: {ssh_url}, error: {e}"
131                    ))),
132                }
133            })
134            .await?
135        };
136
137        // Match outside of the spawn_blocking to avoid ownership issues
138        match result {
139            Ok(m) => {
140                if let Some(t) = self.output.get() {
141                    if *t == OutputMode::Print && self.is_tty {
142                        println!("{m}");
143                    }
144                }
145                self.append_slack_message(m).await;
146            }
147            Err(e) => {
148                if let Some(t) = self.output.get() {
149                    if *t == OutputMode::Print && self.is_tty {
150                        eprintln!("{e}");
151                    }
152                }
153                self.append_slack_error(e.to_string()).await;
154                return Err(e);
155            }
156        }
157
158        Ok(())
159    }
160    pub async fn backup_all_repos(
161        &self,
162        repositories: Vec<String>,
163        path: &Path,
164    ) -> Result<(), GitError> {
165        let mut futures = FuturesUnordered::new();
166        let number_of_repos = repositories.len();
167
168        if self.is_tty && number_of_repos > 1 {
169            let _ = self.output.set(OutputMode::Progress);
170        }
171
172        let progress = if self.is_tty {
173            let pb = ProgressBar::new(number_of_repos as u64);
174            pb.set_style(
175                ProgressStyle::with_template(
176                    "[{elapsed_precise}] [{bar:40.cyan/blue}] {pos}/{len} {msg}",
177                )
178                .unwrap()
179                .progress_chars("#>-"),
180            );
181            pb.enable_steady_tick(std::time::Duration::from_millis(100));
182            Some(Arc::new(pb))
183        } else {
184            None
185        };
186
187        if let Some(ref progress) = progress {
188            progress.println("Backing up all configured repositories...");
189        }
190
191        for repo in repositories {
192            //let progress = progress.clone();
193            futures.push(async move {
194                let result = self.backup_repo(repo.clone(), path).await;
195                (result, repo)
196            });
197        }
198
199        let mut errors: Vec<(String, GitError)> = Vec::new();
200        while let Some((result, repo)) = futures.next().await {
201            if let Some(ref progress) = progress {
202                progress.inc(1);
203            }
204            match result {
205                Ok(()) => {
206                    self.append_slack_message(format!("Backed up repository: {repo}"))
207                        .await;
208
209                    if let Some(ref progress) = progress {
210                        progress.println(format!("Backed up repository: {repo}"));
211                    }
212                }
213                Err(e) => {
214                    self.append_slack_error(format!(
215                        "Failed to back up repository: {repo}, error: {e}"
216                    ))
217                    .await;
218
219                    if let Some(ref progress) = progress {
220                        progress.println(format!("Failed to back up repository: {repo}"));
221                    }
222                    errors.push((repo, e));
223                }
224            }
225        }
226        if let Some(progress) = progress {
227            progress.finish();
228        }
229        Ok(())
230    }
231    pub async fn prune_backup(&self, path: &Path, since: Option<String>) -> Result<(), GitError> {
232        let ext = match path.extension() {
233            Some(ext) => ext.to_str().unwrap_or(""),
234            _ => return Err(GitError::Other("Failed to get file extension".to_string())),
235        };
236
237        if !path.exists() {
238            return Err(GitError::Other(format!(
239                "Path {} does not exist",
240                path.display()
241            )));
242        }
243        if ext != "git" {
244            return Err(GitError::Other(format!(
245                "Path {} is not a git repository",
246                path.display()
247            )));
248        }
249        let path = path.to_path_buf();
250        let result = tokio::task::spawn_blocking(move || {
251            let output = if let Some(since) = since {
252                Command::new("git")
253                    .stdout(Stdio::piped())
254                    .stderr(Stdio::piped())
255                    .args(["gc", &format!("--prune={since}")])
256                    .current_dir(&path)
257                    .output()
258            } else {
259                Command::new("git")
260                    .stdout(Stdio::piped())
261                    .stderr(Stdio::piped())
262                    .args(["gc", "--prune=now"])
263                    .current_dir(&path)
264                    .output()
265            };
266
267            match output {
268                // This branch happens only if the output of the command was successful
269                Ok(s) if s.status.success() => Ok::<_, GitError>(format!(
270                    "Successfully pruned backup at path: {}",
271                    path.display()
272                )),
273                Ok(s) => {
274                    let stderr = String::from_utf8_lossy(&s.stderr);
275                    eprintln!("Error while pruning backup.");
276                    eprintln!("STDERR:\n{stderr}");
277                    Err(GitError::Other(format!(
278                        "Failed to prune backup at path: {}, git exited with status: {}",
279                        path.display(),
280                        s.status
281                    )))
282                }
283                Err(e) => Err(GitError::Other(format!(
284                    "Failed to prune backup at path: {}, error: {e}",
285                    path.display()
286                ))),
287            }
288        })
289        .await?;
290
291        match result {
292            Ok(m) => {
293                if let Some(t) = self.output.get() {
294                    if *t == OutputMode::Print && self.is_tty {
295                        println!("{m}");
296                    }
297                }
298                self.append_slack_message(m).await;
299            }
300            Err(e) => {
301                if let Some(t) = self.output.get() {
302                    if *t == OutputMode::Print && self.is_tty {
303                        eprintln!("{e}");
304                    }
305                }
306                self.append_slack_error(e.to_string()).await;
307                return Err(e);
308            }
309        }
310
311        Ok(())
312    }

314    /// Upload a backup to s3. This requires your aws credentials to be configured in the
315    /// environment
316    pub async fn backup_to_s3(
317        &self,
318        file_path: &Path,
319        bucket: &str,
320        key: &str,
321    ) -> Result<(), GitError> {
322        let region = RegionProviderChain::default_provider().or_else("ca-central-1");
323
324        // Use the behavour version from when this was developped
325        let config = aws_config::defaults(BehaviorVersion::v2025_08_07())
326            .region(region)
327            .load()
328            .await;
329
330        let client = Arc::new(Client::new(&config));
331
332        let mut file = File::open(file_path).await?;
333        file.set_max_buf_size(PART_SIZE);
334        let mut file_buffer = BufReader::new(file);
335        let mut part_number = 1;
336
337        // If the file is smaller than the minimum size allowed for multipart uploads, just upload
338        // the whole thing
339        let size = file_buffer.get_ref().metadata().await?.len();
340        if size == 0 {
341            return Err(GitError::Other("Backup file is empty".to_string()));
342        }
343
344        if size < PART_SIZE as u64 {
345            let body = aws_sdk_s3::primitives::ByteStream::from_path(file_path)
346                .await
347                .map_err(|e| GitError::Other(e.to_string()))?;
348            client
349                .put_object()
350                .bucket(bucket)
351                .key(key)
352                .content_type("application/gzip")
353                .body(body)
354                .send()
355                .await
356                .map_err(|e| GitError::AWSError(e.to_string()))?;
357            return Ok(());
358        }
359
360        // Initiate the multipart upload
361        let response = client
362            .create_multipart_upload()
363            .bucket(bucket)
364            .key(key)
365            .content_type("application/gzip")
366            .send()
367            .await
368            .map_err(|e| GitError::AWSError(e.to_string()))?;
369
370        let upload_id = response
371            .upload_id()
372            .ok_or_else(|| GitError::AWSError("Failed to get upload ID".to_string()))?;
373
374        let mut futures: FuturesUnordered<JoinHandle<Result<(i32, String), _>>> =
375            FuturesUnordered::new();
376
377        let mut results: Vec<(i32, String)> = Vec::new();
378
379        loop {
380            let mut buffer = vec![0u8; PART_SIZE];
381
382            let n = file_buffer.read(&mut buffer).await?;
383            if n == 0 {
384                break;
385            }
386
387            buffer.truncate(n);
388            let client = Arc::clone(&client);
389            let bucket = bucket.to_string();
390            let key = key.to_string();
391            let upload_id = upload_id.to_string();
392            let local_part_number = part_number;
393
394            let lock = Arc::clone(&self.semaphore).acquire_owned().await?;
395            println!("Starting upload of part {local_part_number}");
396            futures.push(tokio::spawn(async move {
397                let _lock = lock;
398                let response = client
399                    .upload_part()
400                    .bucket(&bucket)
401                    .key(&key)
402                    .upload_id(&upload_id)
403                    .part_number(local_part_number)
404                    .body(buffer.into())
405                    .send()
406                    .await
407                    .map_err(|e| GitError::AWSError(e.to_string()))?;
408                let etag = response.e_tag().unwrap().to_string();
409                println!("Uploaded part {local_part_number}");
410
411                Ok::<(i32, String), GitError>((local_part_number, etag))
412            }));
413            part_number += 1;
414        }
415        while let Some(res) = futures.next().await {
416            match res {
417                Ok(r) => results.push(r?),
418                Err(e) => {
419                    return Err(GitError::Other(format!(
420                        "Failure when uploading multipart file: {e}"
421                    )));
422                }
423            }
424        }
425        results.sort_by_key(|part| part.0);
426
427        let completed_parts: Vec<CompletedPart> = results
428            .into_iter()
429            .map(|(part_number, etag)| {
430                CompletedPart::builder()
431                    .set_part_number(Some(part_number))
432                    .set_e_tag(Some(etag))
433                    .build()
434            })
435            .collect();
436
437        let completed_upload = CompletedMultipartUpload::builder()
438            .set_parts(Some(completed_parts))
439            .build();
440
441        let response = client
442            .complete_multipart_upload()
443            .bucket(bucket)
444            .key(key)
445            .upload_id(upload_id)
446            .multipart_upload(completed_upload)
447            .send()
448            .await
449            .map_err(|e| GitError::AWSError(e.to_string()))?;
450
451        println!("Uploaded to {:?}", response.location().unwrap_or_default());
452
453        Ok(())
454    }
