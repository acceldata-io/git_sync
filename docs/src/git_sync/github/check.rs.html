31impl GithubClient {
32    /// Get branches that are older than a certain number of days
33    /// A blacklist should be passed to ignore certain branches, since some of them are static and
34    /// will never change, nor should they be deleted.
35    #[allow(clippy::too_many_lines)]
36    pub async fn check_repository(
37        &self,
38        url: &str,
39        blacklist: HashSet<String>,
40        checks: &RepoChecks,
41    ) -> Result<Checks, GitError> {
42        let info = get_repo_info_from_url(url)?;
43        let (owner, repo) = (info.owner, info.repo_name);
44        let octocrab = self.octocrab.clone();
45        let (get_branches, days_ago) = checks.old_branches;
46        let get_protection = checks.protected;
47        let get_license = checks.license;
48        let branch_filter = checks.branch_filter.clone();
49
50        let mut protection_rules: Vec<BranchProtectionRule> = Vec::new();
51        let mut license: Option<LicenseInfo>;
52
53        // Use a graphql query to drastically reduce the number of api calls we need to make.
54        // This lets us get the repo name and latest commit date in one call, instead of two.
55        // This makes a huge difference for very large repositories with many branches.
56        let query = r#"
57            query($owner: String!, $repo: String!, $after: String, $getLicenseInfo: Boolean!, $getBranches: Boolean!, $getProtection: Boolean!) {
58                repository(owner: $owner, name: $repo) {
59                    licenseInfo @include(if: $getLicenseInfo) {
60                        name
61                        spdxId
62                        url
63                    }
64                    branchProtectionRules(first: 100) @include(if: $getProtection){
65                        nodes {
66                            pattern
67                            isAdminEnforced
68                            requiresApprovingReviews
69                            requiredApprovingReviewCount
70                            requiresStatusChecks
71                            requiresStrictStatusChecks
72                            requiresConversationResolution
73                            restrictsPushes
74                            restrictsReviewDismissals
75                        }
76                    }
77
78                    refs(refPrefix: "refs/heads/", first: 100, after: $after) @include(if: $getBranches) {
79                        nodes {
80                            name
81                            target {
82                                ... on Commit {
83                                    committedDate
84                                }
85                            }
86                        }
87                        pageInfo {
88                            hasNextPage
89                            endCursor
90                        }
91                    }
92                }
93            }
94        "#;
95
96        let mut branches = HashMap::<String, DateTime<Utc>>::new();
97
98        // Needed for the graphql query
99        let mut after: Option<String> = None;
100
101        loop {
102            // Acquire a lock on the semaphore
103            let permit = self.semaphore.clone().acquire_owned().await?;
104
105            // Paginated results, so we have to loop over until there aren't any more pages left
106            let payload = serde_json::json!({
107                "query": query,
108                "variables": {
109                    "owner": owner,
110                    "repo": repo,
111                    "after": after,
112                    "getBranches": get_branches,
113                    "getProtection": get_protection,
114                    "getLicenseInfo": get_license,
115                }
116            });
117            let response: serde_json::Value = async_retry!(
118                ms = 100,
119                timeout = 5000,
120                retries = 3,
121                error_predicate = |e: &octocrab::Error| is_retryable(e),
122                body = { octocrab.graphql(&payload).await },
123            )?;
124            drop(permit);
125            let refs = &response["data"]["repository"]["refs"];
126            license = response["data"]["repository"]["licenseInfo"]
127                .as_object()
128                .and_then(|license| {
129                    serde_json::from_value::<LicenseInfo>(serde_json::Value::Object(
130                        license.clone(),
131                    ))
132                    .ok()
133                });
134
135            if let Some(nodes) = refs["nodes"].as_array() {
136                for branch in nodes {
137                    if let Some(name) = branch.get("name").and_then(|v| v.as_str()) {
138                        if let Some(re) = &branch_filter
139                            && !re.is_match(name)?
140                        {
141                            continue;
142                        }
143
144                        let date_str = branch
145                            .pointer("/target/committedDate")
146                            .and_then(|v| v.as_str())
147                            .unwrap_or("");
148                        let date = date_str.parse::<DateTime<Utc>>();
149                        match date {
150                            Ok(d) => {
151                                branches.insert(name.to_string(), d);
152                            }
153                            Err(e) => {
154                                eprintln!(
155                                    "Failed to parse date for {owner}/{repo} branch {name}: {e}"
156                                );
157                            }
158                        }
159                    }
160                }
161            }
162            if let Some(protection) =
163                response["data"]["repository"]["branchProtectionRules"]["nodes"].as_array()
164            {
165                for rule in protection {
166                    let parsed_rule: Option<BranchProtectionRule> =
167                        serde_json::from_value(rule.clone()).ok();
168                    if let Some(r) = parsed_rule {
169                        protection_rules.push(r);
170                    }
171                }
172            }
173
174            let page_info = &refs["pageInfo"];
175            let has_next_page = page_info["hasNextPage"].as_bool().unwrap_or(false);
176            after = page_info["endCursor"]
177                .as_str()
178                .map(std::string::ToString::to_string);
179
180            if !has_next_page {
181                break;
182            }
183        }
184        let now = Utc::now();
185
186        let mut old_branches: Vec<_> = branches
187            .into_iter()
188            .filter(|(_, age)| (now - age).num_days() >= days_ago)
189            .filter(|(branch, _)| !blacklist.contains(branch))
190            .collect();
191        old_branches.sort_by_key(|(_, age)| *age);
192
193        let old_branches: Vec<(String, String)> = old_branches
194            .into_iter()
195            .map(|(branch, age)| {
196                let formatted = age.format("%Y-%m-%d").to_string();
197                (branch.clone(), formatted.clone())
198            })
199            .collect();
200
201        Ok(Checks {
202            branches: old_branches,
203            rules: protection_rules,
204            license,
205            repo: format!("{owner}/{repo}"),
206        })
207    }
208    /// Display the results of the various checks into a nice table
209    pub fn display_check_results(
210        &self,
211        header: Vec<String>,
212        rows: Vec<Vec<String>>,
213        rules: &[BranchProtectionRule],
214        license: Option<&LicenseInfo>,
215        repo: &str,
216    ) {
217        if self.is_tty {
218            println!("TTY?");
219            let table = Table::builder(tabled::settings::style::Style::ascii())
220                .title(format!("Stale Branches for {repo}"))
221                .header(header)
222                .rows(rows)
223                .centre(false)
224                .align(tabled::settings::Alignment::center())
225                .build();
226            println!("{table}");
227            if !rules.is_empty() {
228                for rule in rules {
229                    println!("{rule}");
230                }
231            }
232            if let Some(license) = &license
233                && let Some(name) = &license.name
234            {
235                println!("License: {name}");
236            }
237        } else {
238            let mut output = repo.to_string();
239            let license_name = if let Some(license) = &license
240                && let Some(name) = &license.name
241            {
242                format!(",{name}")
243            } else {
244                String::new()
245            };
246            for v in rows {
247                let line = v.join(",");
248                let _ = write!(output, "{line}");
249                let line = format!("{repo},{line},{license_name}");
250                println!("{line}");
251            }
252            if let Some(license) = &license
253                && let Some(name) = &license.name
254            {
255                let _ = write!(output, ",{name}");
256            }
257        }
258    }
259    /// Check the results and send any errors to slack. None of our repositories currently have any
260    /// rules
261    pub async fn validate_check_results(
262        &self,
263        repository: &str,
264        check: Checks,
265        branch_blacklist: HashSet<String>,
266        license_blacklist: HashSet<String>,
267    ) -> Result<(), GitError> {
268        if let Some(l) = &check.license
269            && let Some(spdx) = &l.spdx_id
270            && license_blacklist.contains(spdx)
271        {
272            self.append_slack_error(format!("Blacklisted license {spdx} used in {repository}"))
273                .await;
274        }
275
276        if !check.branches.is_empty() {
277            let branches: Vec<(String, String)> = check
278                .branches
279                .into_iter()
280                .filter(|(branch, _)| !branch_blacklist.contains(branch))
281                .collect();
282            let num_branches = branches.len();
283            if num_branches > 0 {
284                self.append_slack_error(format!(
285                    "Repository {repository} has {num_branches} stale branches"
286                ))
287                .await;
288            }
289        }
290
291        Ok(())
292    }
293    /// Run the selected checks against all configured repositories
294    pub async fn check_all_repositories(
295        &self,
296        repositories: Vec<String>,
297        checks: &RepoChecks,
298        blacklist: &HashSet<String>,
299    ) -> Result<Vec<Checks>, GitError> {
300        let mut futures = FuturesUnordered::new();
301        for repo in repositories {
302            futures.push(async move {
303                let response = self
304                    .check_repository(&repo, blacklist.clone(), checks)
305                    .await;
306                (response, repo)
307            });
308        }
309
310        let mut results: Vec<Checks> = Vec::new();
311        let mut errors: Vec<(String, GitError)> = Vec::new();
312        while let Some((result, repo)) = futures.next().await {
313            match result {
314                Ok(r) => results.push(r),
315                Err(e) => errors.push((repo, e)),
316            }
317        }
318        if errors.is_empty() {
319            Ok(results)
320        } else {
321            Err(GitError::MultipleErrors(errors))
322        }
323    }
