72/// Create a sample config file
73pub fn generate_config(path: Option<&PathBuf>, force: bool) -> Result<PathBuf, GitError> {
74    // This is only needed if a path isn't provided
75    let xdg = Xdg::new();
76    let config_path = if let Ok(xdg) = xdg {
77        let f = xdg.config_file(CONFIG_NAME);
78        f.ok()
79    } else {
80        None
81    };
82
83    match (path, force, config_path) {
84        (Some(path), true, _) => {
85            if path.exists() {
86                let mut backup = path.clone().into_os_string();
87                backup.push(".bak");
88                fs::copy(path, backup)?;
89            }
90            fs::write(path, SAMPLE_CONFIG)?;
91            Ok(path.clone())
92        }
93        (Some(path), false, _) => {
94            if path.exists() {
95                return Err(GitError::Other(format!(
96                    "Config file already exists at {}. Use --force to overwrite",
97                    path.display()
98                )));
99            }
100            fs::write(path, SAMPLE_CONFIG)?;
101            Ok(path.clone())
102        }
103        (None, true, Some(config_path)) => {
104            if config_path.exists() {
105                let mut backup = config_path.clone().into_os_string();
106                backup.push(".bak");
107                fs::copy(&config_path, backup)?;
108            }
109            fs::write(&config_path, SAMPLE_CONFIG)?;
110            Ok(config_path)
111        }
112        (None, false, Some(config_path)) => {
113            println!("{}", config_path.display());
114            if config_path.exists() {
115                return Err(GitError::Other(format!(
116                    "Config file already exists at {}. Use --force to overwrite",
117                    config_path.display()
118                )));
119            }
120            fs::write(&config_path, SAMPLE_CONFIG)?;
121            Ok(config_path)
122        }
123        (None, _, None) => {
124            panic!(
125                "No config file specified and $XDG_CONFIG_HOME or $HOME/.config could not be determined. Cannot continue"
126            );
127        }
128    }
