24/// A basic sample configuration that can be initialized
25/// by using the `config` command
26const SAMPLE_CONFIG: &str = r#"#Git repo syncing configuration
27[github]
28#token = "Add your github api token here"
29
30# Needed if you want slack integration
31# webhook_url should be an https address
32[slack]
33# Eventually, the enabled option will be added to always enable slack for each run
34#enable=true
35webhook_url = ""
36
37# This can be used to change who is commiting changes and creating
38# releases, instead of using the user who is already set up for git
39# in the environment.
40[info]
41#name = "set the name of who is commiting changes"
42#email = "set the email of who is commiting changes"
43
44# All repositories should be an https url to the repository
45# Any operation that needs an ssh url will be automatically
46# converted.
47[repos]
48#fork = ["repo1", "repo2"]
49# This is for forked repositories that do not have a configured parent repository.
50# This is a map of {"forked_url": "real_upstream_url"}
51#fork_with_workaround = {}
52#private = ["repo1", "repo2"]
53#public = []
54
55# You can also have any number of arbitrary groups:
56#my-group = ["repo3", "repo10"]
57

66license_blacklist = [

71    "GPL-2.0-only",

75/// Create a sample config file
76pub fn generate_config(path: Option<&PathBuf>, force: bool) -> Result<PathBuf, GitError> {
77    // This is only needed if a path isn't provided
78    let xdg = Xdg::new();
79    let config_path = if let Ok(xdg) = xdg {
80        let f = xdg.config_file(CONFIG_NAME);
81        f.ok()
82    } else {
83        None
84    };
85
86    match (path, force, config_path) {
87        (Some(path), true, _) => {
88            if path.exists() {
89                let mut backup = path.clone().into_os_string();
90                backup.push(".bak");
91                fs::copy(path, backup)?;
92            }
93            fs::write(path, SAMPLE_CONFIG)?;
94            Ok(path.clone())
95        }
96        (Some(path), false, _) => {
97            if path.exists() {
98                return Err(GitError::Other(format!(
99                    "Config file already exists at {}. Use --force to overwrite",
100                    path.display()
101                )));
102            }
103            fs::write(path, SAMPLE_CONFIG)?;
104            Ok(path.clone())
105        }
106        (None, true, Some(config_path)) => {
107            if config_path.exists() {
108                let mut backup = config_path.clone().into_os_string();
109                backup.push(".bak");
110                fs::copy(&config_path, backup)?;
111            }
112            fs::write(&config_path, SAMPLE_CONFIG)?;
113            Ok(config_path)
114        }
115        (None, false, Some(config_path)) => {
116            println!("{}", config_path.display());
117            if config_path.exists() {
118                return Err(GitError::Other(format!(
119                    "Config file already exists at {}. Use --force to overwrite",
120                    config_path.display()
121                )));
122            }
123            fs::write(&config_path, SAMPLE_CONFIG)?;
124            Ok(config_path)
125        }
126        (None, _, None) => {
127            panic!(
128                "No config file specified and $XDG_CONFIG_HOME or $HOME/.config could not be determined. Cannot continue"
129            );
130        }
131    }
132}
