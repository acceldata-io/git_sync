22//! Manage Github repositories from the command line.
23//! You can perform an action on one repository, or on all repositories that are configured in the
24//! configuration file.
25//!
26//! This tool is still in development so some features may not work as expected
27//!
28//! You can do the following:
29//! 1. Sync a forked repository with its upstream parent repository.
30//! 2. Sync tags from its parent repository
31//! 3. Manage branches (create, delete)
32//! 4. Manage tags (create, delete)
33//! 5. Create releases with automatically generated release notes
34//! 6. Create and merge pull requests
35//! 7. Run various sanity checks on repositories
36//! 8. Create a PR and automatically merge it, if possible
37//! 9. Backup a repository, including to S3 if enabled at build time
38//! 10. Send notifications to slack if the slack feature is enabled
39//! 11. Do all of the above for a single repository, or for all configured repositories
40//!
41//! Some examples of how to use this tool:
42//! ```shell
43//! git_sync tag create --tag v1.0.2 --branch my_dev_branch --repo https://github.com/org/my_repository
44//! git_sync tag create --tag v1.0.0 --branch my_common_dev_branch --all --slack # will use all repos in your config file and enable slack notifications
45//! ```
46//! When running a command with --all, you can specify which category of repositories to run the
47//! command on by specifying the `--repository` argument (valid options are 'fork', 'private',
48//! 'public', or 'all').
49//!
50//! You can generate manpages and shell completions using the generate command:
51//! ```shell
52//! git_sync generate --kind man
53//! git_sync generate --kind bash # or fish, or zsh
54//! ```
55//!
56//! When building this tool, you should use
57//! ```shell
58//! cargo auditable build --release
59//! ```
60//!
61//! [cargo-auditable](https://github.com/rust-secure-code/cargo-auditable) and [cargo-audit](https://github.com/RustSec/rustsec/tree/main/cargo-audit) can be installed by running
62//! ```shell
63//! cargo install cargo-audit cargo-auditable
64//! ```
65//!
66//! This way the binary will be built with information about the build dependencies embedded into
67//! the binary so that it can be scanned with tools like `cargo-audit` or Trivy.

76use cli::parse_args;
77use config::Config;
78use error::GitError;
79use github::match_args;
80use rustls::crypto::aws_lc_rs;
81#[tokio::main]
82async fn main() -> Result<(), GitError> {
83    // This needs to be here to make sure we're using aws-lc-rs.
84    // aws-lc-rs is actively being developed.
85    // One or two crates default to ring, so we override it here.
86    let provider = aws_lc_rs::default_provider().install_default();
87    if provider.is_err() {
88        return Err(GitError::Other(
89            "Failed to install AWS-LC-RS as default TLS provider".to_string(),
90        ));
91    }
92    let args = parse_args();
93
94    let result: Result<(), GitError> = {
95        let config = Config::new(args.file.as_ref())?;
96        match_args::match_arguments(&args, config).await
97    };
98    // Get nice error messages, with simple suggestions instead of huge structs
99    if let Err(e) = result {
100        if e.is_broken_pipe() {
101            std::process::exit(0)
102        }
103        let error = e.to_user_error();
104        eprintln!("{error}");
105        std::process::exit(1);
106    }
107
108    Ok(())
109}
