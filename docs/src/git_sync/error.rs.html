91impl GitError {
92    /// Convert this error into a user-friendly version
93    #[allow(clippy::too_many_lines)]
94    pub fn to_user_error(&self) -> UserError {
95        match self {
96            GitError::GithubApiError(e) => {
97                let (code, msg) = octocrab_error_info(e);
98                let category = if is_network_error(e) {
99                    ErrorCategory::Network
100                } else{
101                    match code {
102                        Some(http::StatusCode::UNAUTHORIZED | http::StatusCode::FORBIDDEN) => ErrorCategory::Auth,
103                        Some(http::StatusCode::NOT_FOUND) => ErrorCategory::NotFound,
104                        Some(http::StatusCode::TOO_MANY_REQUESTS) => ErrorCategory::RateLimit,
105                        Some(ref c) if c.is_server_error() => ErrorCategory::Server,
106                        Some(ref c) if c.is_client_error() => ErrorCategory::Input,
107                        _ => ErrorCategory::Unknown,
108                    }
109                };
110                let suggestion = match category {
111                    ErrorCategory::Auth => Some("Check your credentials and repository permissions.".into()),
112                    ErrorCategory::RateLimit => Some("You are being rate-limited by GitHub. Wait and try again later.".into()),
113                    ErrorCategory::Network => Some("Check your Internet connection.".into()),
114                    ErrorCategory::NotFound => Some("Check the repository or resource name.".into()),
115                    _ => None,
116                };
117                UserError { code, message: msg, suggestion }
118            }
119            GitError::PRNotMergeable(pr_number) => UserError {
120                code: None,
121                message: format!("PR #{pr_number} cannot be merged automatically"),
122                suggestion: Some("Human intervention may be required.".into()),
123            },
124            GitError::NoUpstreamRepo => UserError {
125                code: None,
126                message: "Could not find an upstream repository for this fork.".into(),
127                suggestion: Some("Make sure the repository is a fork and has an upstream set.".into()),
128            },
129            GitError::DateParseError(e) => UserError {
130                code: None,
131                message: format!("Invalid date: {e}"),
132                suggestion: Some("Use a valid date format.".into()),
133            },
134            GitError::InvalidRepository(url) => UserError {
135                code: None,
136                message: format!("Invalid repository URL: {url}"),
137                suggestion: Some("Check the repository URL.".into()),
138            },
139            GitError::RegexError(e) => UserError {
140                code: None,
141                message: format!("Regex error: {e}"),
142                suggestion: Some("Check your regular expression syntax.".into()),
143            },
144            GitError::NoReposConfigured => UserError {
145                code: None,
146                message: "No repositories are configured.".into(),
147                suggestion: Some("Add repositories to your configuration.".into()),
148            },
149            GitError::NotAFork => UserError {
150                code: None,
151                message: "Repository is not a fork.".into(),
152                suggestion: Some("Choose a repository that is a fork.".into()),
153            },
154            GitError::MissingRepositoryName => UserError {
155                code: None,
156                message: "Repository name is missing.".into(),
157                suggestion: Some("Specify the repository name.".into()),
158            },
159            GitError::NoSuchBranch(branch) => UserError {
160                code: None,
161                message: format!("No such branch: {branch}"),
162                suggestion: Some("Check the branch name or create it first.".into()),
163            },
164            GitError::NoSuchTag(tag) => UserError {
165                code: None,
166                message: format!("No such tag: {tag}"),
167                suggestion: Some("Check the tag name or create it first.".into()),
168            },
169            GitError::TomlError(e) => UserError {
170                code: None,
171                message: format!("TOML parsing error: {e}"),
172                suggestion: Some("Check your config file for syntax errors.".into()),
173            },
174            GitError::MissingToken => UserError {
175                code: None,
176                message: "Missing GitHub token. Please provide a token via --token, GITHUB_TOKEN environment variable, or in your config file.".into(),
177                suggestion: Some("Provide a GitHub token using --token, environment variable, or config file.".into()),
178            },
179            GitError::JoinError(e) => UserError {
180                code: None,
181                message: format!("Failed to join tasks: {e}"),
182                suggestion: None,
183            },
184            #[cfg(feature = "aws")]
185            GitError::AWSError(e) => UserError {
186                code: None,
187                message: format!("AWS Error: {e}"),
188                suggestion: None,
189            },
190            GitError::IoError(e) => UserError {
191                code: None,
192                message: format!("IO error: {e}"),
193                suggestion: Some("Check file paths and permissions, or try again.".into()),
194            },
195            GitError::JsonError(e) => UserError {
196                code: None,
197                message: format!("JSON error: {e}"),
198                suggestion: Some("Check your JSON syntax.".into()),
199            },
200            GitError::GetTimezoneError(e) => UserError {
201                code: None,
202                message: format!("Timezone error: {e}"),
203                suggestion: Some("Check your timezone string or system settings.".into()),
204            },
205            GitError::SemaphoreError(e) => UserError {
206                code: None,
207                message: format!("Semaphore Error: {e}"),
208                suggestion: None,
209            },
210
211            GitError::Other(msg) => UserError {
212                code: None,
213                message: msg.clone(),
214                suggestion: None,
215            },
216            GitError::MultipleErrors(errors) => {
217                // Show the first error, but mention there were multiple
218                if errors.is_empty() {
219                    UserError {
220                        code: None,
221                        message: "Multiple unknown errors occurred.".into(),
222                        suggestion: None,
223                    }
224                } else  {
225                    let mut message = "Multiple errors: ".to_string();
226                    for (context, err) in errors {
227                        let e = err.to_user_error();
228                        let _ = writeln!(message, "\n\t- {context}: {}", e.message);
229                    }
230                    UserError {
231                        code: None,
232                        message,
233                        suggestion: None,
234                    }
235                }
236            }
237        }
238    }
239    /// Walk through the error chain to find out if it's a broken pipe
240    pub fn is_broken_pipe(&self) -> bool {
241        let mut source = Some(self as &dyn std::error::Error);
242        while let Some(err) = source {
243            if let Some(ioe) = err.downcast_ref::<std::io::Error>() {
244                if ioe.kind() == std::io::ErrorKind::BrokenPipe {
245                    return true;
246                }
247            }
248            source = err.source();
249        }
250        false
251    }
252}

271/// Holds information about an error to present to the user
272#[derive(Debug)]
273pub struct UserError {
274    pub code: Option<http::StatusCode>,
275    pub message: String,
276    pub suggestion: Option<String>,
277}
278
279impl std::fmt::Display for UserError {
280    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
281        if let Some(code) = self.code {
282            let message = format!("HTTP error ({code:?}): {}", self.message);
283            let formatted = format!("{}", style(message).red().bold());
284            writeln!(f, "{formatted}")?;
285        } else {
286            let message = format!("Error: {}", self.message);
287            writeln!(f, "{}", style(message).red().bold())?;
288        }
289
290        if let Some(suggestion) = &self.suggestion {
291            let message = format!("Suggestion: {suggestion}");
292            write!(f, "{}", style(message).green())?;
293        }
294        Ok(())
295    }
296}

298/// Check if whatever error is passed is reasonably retryable. Most errors aren't.
299pub fn is_retryable(e: &octocrab::Error) -> bool {
300    match e {
301        octocrab::Error::GitHub { source, .. } => {
302            // If either of these are true, then we
303            let msg = &source.message;
304            let status = source.status_code;
305
306            if status.is_server_error() {
307                return true;
308            } else if status == http::StatusCode::FORBIDDEN {
309                // Sometimes a 403 is returned for rate limiting
310                if msg.contains("rate limit") || msg.contains("abuse detection") {
311                    eprintln!("Rate limited by GitHub API: {msg}");
312                }
313                return false;
314            } else if status == http::StatusCode::TOO_MANY_REQUESTS {
315                eprintln!("Rate limited by GitHub API: {msg}");
316                return false;
317            } else if status.is_client_error() {
318                return false;
319            }
320            false
321        }
322        octocrab::Error::Http { .. }
323        | octocrab::Error::Hyper { .. }
324        | octocrab::Error::Service { .. } => true,
325        _ => false,
326    }
327}

329/// Helper function to extract errors from an `octocrab::Error`
330pub fn octocrab_error_info(e: &octocrab::Error) -> (Option<http::StatusCode>, String) {
331    match e {
332        octocrab::Error::GitHub { source, .. } => {
333            let status = source.status_code;
334            let mut message = source.message.to_string();
335            if let Some(errors) = &source.errors {
336                for err in errors {
337                    if let Some(obj) = err.as_object() {
338                        if let Some(msg) = obj.get("message").and_then(|m| m.as_str()) {
339                            let m = format!("\n- {msg}");
340                            let _ = write!(message, "{m}");
341                        }
342                    }
343                }
344            }
345            (Some(status), message.to_string())
346        }
347        octocrab::Error::Http { source, .. } => (None, source.to_string()),
348        octocrab::Error::UriParse { source, .. } => (None, source.to_string()),
349        octocrab::Error::Uri { source, .. } => (None, source.to_string()),
350        octocrab::Error::Installation { .. } => (None, "Installation error".to_string()),
351        octocrab::Error::InvalidHeaderValue { source, .. } => (None, source.to_string()),
352        octocrab::Error::InvalidUtf8 { source, .. } => (None, source.to_string()),
353        octocrab::Error::Encoder { source, .. } => (None, source.to_string()),
354        octocrab::Error::Hyper { source, .. } => (None, source.to_string()),
355        octocrab::Error::SerdeUrlEncoded { source, .. } => (None, source.to_string()),
356        octocrab::Error::Serde { source, .. } => (None, source.to_string()),
357        octocrab::Error::Json { source, .. } => (None, source.to_string()),
358        octocrab::Error::JWT { source, .. } => (None, source.to_string()),
359        octocrab::Error::Other { source, .. } | octocrab::Error::Service { source, .. } => {
360            (None, source.to_string())
361        }
362        _ => (None, "Unknown error".to_string()),
363    }
364}

366fn is_network_error(e: &(dyn Error + 'static)) -> bool {
367    let mut current = Some(e);
368
369    while let Some(err) = current {
370        if let Some(hyper_err) = err.downcast_ref::<hyper::Error>() {
371            if hyper_err.is_closed() || hyper_err.is_incomplete_message() || hyper_err.is_timeout()
372            {
373                return true;
374            }
375        }
376
377        // reqwest errors (timeout, connect, etc.)
378        if let Some(reqwest_err) = err.downcast_ref::<reqwest::Error>() {
379            if reqwest_err.is_timeout() || reqwest_err.is_connect() || reqwest_err.is_request() {
380                return true;
381            }
382        }
383
384        if let Some(io_err) = err.downcast_ref::<std::io::Error>() {
385            match io_err.kind() {
386                TimedOut | ConnectionRefused | ConnectionAborted | ConnectionReset
387                | NotConnected | AddrNotAvailable | AddrInUse | NetworkUnreachable
388                | HostUnreachable | BrokenPipe | WouldBlock => return true,
389                _ => {}
390            }
391        }
392
393        // String-based fallback (last resort)
394        let msg = err.to_string().to_lowercase();
395        if msg.contains("dns error")
396            || msg.contains("failed to lookup address")
397            || msg.contains("nodename nor servname provided")
398            || msg.contains("network unreachable")
399            || msg.contains("no route to host")
400            || msg.contains("connection refused")
401            || msg.contains("timed out")
402        {
403            return true;
404        }
405
406        current = err.source();
407    }
408    false
